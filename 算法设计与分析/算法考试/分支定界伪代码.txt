分支（线路e，是左子树还是右子树）
begin
	if (是左子树)
	begin
		add 线路e to 选择的线路集合；
		if（选择的线路集合中线路条数＞49 or 选择的线路集合会形成环 or 当前敷设费用＞最小费用 or 当前总地井数＞UMAX or 当前跨区线路数＞DMAX） return；
		if （选择的线路集合中线路条数为49 && 这些线路是联通的）
		begin
			最小费用 = 当前敷设费用；
			最优方案 = 选择的线路集合；
			return；
		end；
	end；
	线路next = 线路e的下一条未访问的线路；
	线路next访问标记 = 已访问；
	
	当前敷设费用 += 线路next的敷设费用；
	当前总地井数 += 敷设线路next需要建设的地井数；
	if（线路next是跨区线路） 当前跨区线路数++；
	分支（线路next，左子树）；
	当前敷设费用 -= 线路next的敷设费用；
	当前总地井数 -= 敷设线路next需要建设的地井数；
	if（线路next是跨区线路） 当前跨区线路数--；
				
	分支（线路next， 右子树）；
	
	线路next访问标记 = 未访问；
end；
